95 newAlignment* Phylip32State::LoadAlignment(std::string filename)
96 {
97  /* PHYLIP 3.2 (Interleaved) file format parser */
98  newAlignment* _alignment = new newAlignment();
99 
100  int i, blocksFirstLine, firstLine = true;
101  char *str, *line = NULL;
102  ifstream file;
103 
104  /* Check the file and its content */
105  file.open(filename, ifstream::in);
106  if(!utils::checkFile(file))
107  return nullptr;
108 
109  /* Store the file name for futher format conversion*/
110  _alignment->filename.append("!Title ");
111  _alignment->filename.append(filename);
112  _alignment->filename.append(";");
113 
114  /* Read first valid line in a safer way */
115  do {
116  line = utils::readLine(file);
117  } while ((line == NULL) && (!file.eof()));
118 
119  /* If the file end is reached without a valid line, warn about it */
120  if (file.eof())
121  return nullptr;
122 
123  /* Get the sequences and residues numbers. If there is any mistake,
124  * return a FALSE value to warn about the possible error */
125  str = strtok(line, DELIMITERS);
126  _alignment->sequenNumber = 0;
127  if(str != NULL)
128  _alignment->sequenNumber = atoi(str);
129 
130  str = strtok(NULL, DELIMITERS);
131  _alignment->residNumber = 0;
132  if(str != NULL)
133  _alignment->residNumber = atoi(str);
134 
135  if((_alignment->sequenNumber == 0) || (_alignment->residNumber == 0))
136  return nullptr;
137 
138  /* Reserve memory according to the input parameters */
139  _alignment->sequences = new string[_alignment->sequenNumber];
140  _alignment->seqsName = new string[_alignment->sequenNumber];
141 
142  /* Point to the first sequence in the alignment. Since the alignment could not
143  * have blank lines to separate the different sequences. Store the blocks size
144  * for the first line including a sequence identifier */
145  i = 0;
146  blocksFirstLine = 0;
147 
148  do {
149  /* Read lines in a safer way. Destroy previous stored information */
150  if (line != NULL)
151  delete [] line;
152 
153  line = utils::readLine(file);
154  /* If there is nothing in the input line, skip the loop instructions */
155  if(line == NULL)
156  continue;
157 
158  str = strtok(line, OTHDELIMITERS);
159  /* First block: Sequence Name + Sequence fragment. Count how many blocks
160  * the first sequence line is divided. It could help to identify the
161  * different sequences from the input file */
162  if(firstLine) {
163  _alignment->seqsName[i].append(str, strlen(str));
164  str = strtok(NULL, OTHDELIMITERS);
165  firstLine = 1;
166  }
167 
168  /* Sequence fragment */
169  while(str != NULL) {
170  _alignment->sequences[i].append(str, strlen(str));
171  str = strtok(NULL, OTHDELIMITERS);
172  /* Count the blocks number for the sequences first line */
173  if (firstLine)
174  firstLine += 1;
175  }
176 
177  /* Store the blocks number for the first sequence including the name */
178  if ((blocksFirstLine == 0) and firstLine)
179  blocksFirstLine = firstLine;
180 
181  /* If a false positive new sequence was detected, add stored information for
182  * the current sequence to the previous one and clear the data structure
183  * for the current sequence. Finally, move the sequence pointer to the
184  * previous one. */
185  if ((firstLine != false) and (firstLine != blocksFirstLine)) {
186  _alignment->sequences[i-1].append(_alignment->seqsName[i]);
187  _alignment->seqsName[i].clear();
188  _alignment->sequences[i-1].append(_alignment->sequences[i]);
189  _alignment->sequences[i].clear();
190  i --;
191  }
192 
193  firstLine = false;
194  /* There are many ways to detect a new sequence. */
195  /* One of them -experimental- is just to detect if the residues number for
196  * the current entry is equal to the residues number for the whole align */
197  if ((int) _alignment->sequences[i].size() == _alignment->residNumber) {
198  firstLine = true;
199  i++;
200  }
201  } while(!file.eof());
202 
203  /* Close the input file and delete dinamic memory */
204  file.close();
205  if (line != NULL)
206  delete [] line;
207 
208  /* Check the matrix's content */
209  _alignment->fillMatrices(true);
210  return _alignment;
211 }

213 bool Phylip32State::SaveAlignment(newAlignment* alignment, std::ostream* output, std::string* FileName)
214 {
215  /* Generate output alignment in PHYLIP 3.2 format (interleaved) */
216 
217  int i, j, k, maxLongName;
218  string *tmpMatrix;
219 
220  /* Check whether sequences in the alignment are aligned or not.
221  * Warn about it if there are not aligned. */
222  if (!alignment->isAligned) {
223  cerr << endl << "ERROR: Sequences are not aligned. Format (phylip) "
224  << "not compatible with unaligned sequences." << endl << endl;
225  return false;
226  }
227 
228  /* Allocate local memory for generating output alignment */
229  tmpMatrix = new string[alignment->sequenNumber];
230 
231  /* Depending on alignment orientation: forward or reverse. Copy directly
232  * sequence information or get firstly the reversed sequences and then
233  * copy it into local memory */
234  for(i = 0; i < alignment->sequenNumber; i++)
235  tmpMatrix[i] = (!Machine->reverse) ?
236  alignment->sequences[i] :
237  utils::getReverse(alignment->sequences[i]);
238 
239  /* Depending on if short name flag is activated (limits sequence name up to
240  * 10 characters) or not, get maximum sequence name length */
241  maxLongName = PHYLIPDISTANCE;
242  for(i = 0; (i < alignment->sequenNumber) && (!Machine->shortNames); i++)
243  maxLongName = utils::max(maxLongName, alignment->seqsName[i].size());
244 
245  /* Generating output alignment */
246  /* First Line: Sequences Number & Residued Number */
247  (*output) << " " << alignment->sequenNumber << " " << alignment->residNumber << endl;
248 
249  /* Alignment */
250  /* For each sequence, print its identifier and then the sequence itself in
251  * blocks of 50 residues */
252  for(i = 0; i < alignment->sequenNumber; i++) {
253  /* Sequence Name */
254  (*output) << setw(maxLongName + 3) << left << alignment->seqsName[i].substr(0, maxLongName);
255  /* Sequence. Each line contains a block of 5 times 10 residues. */
256  for(j = 0; j < alignment->residNumber; j += 50) {
257  for(k = j; (k < alignment->residNumber) && (k < (j + 50)); k += 10)
258  (*output) << alignment->sequences[i].substr(k, 10) << " ";
259  (*output) << endl;
260  /* If the sequences end has not been reached, print black spaces
261  * to follow format specifications */
262  if((j + 50) < alignment->residNumber)
263  (*output) << setw(maxLongName + 3) << " ";
264  }
265  /* Print a blank line to mark sequences separation */
266  (*output) << endl;
267  }
268 
269  /* Deallocate local memory */
270  delete [] tmpMatrix;
271 
272  return true;
273 }
