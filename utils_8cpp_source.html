163  return;

302 char* utils::readLine(ifstream &file) {
303  // Read a new line from current input stream. This function is better than
304  // standard one since cares of operative system compability. It is useful
305  // as well because remove tabs and blank spaces at lines beginning/ending
306 
307  int state;
308  char c = ' ';
309  string nline;
310  char *line = NULL;
311 
312  // Check it the end of the file has been reached or not
313  if(file.eof())
314  return NULL;
315 
316  // Store first line found. For -Windows & MacOS compatibility- carriage return
317  // is considered as well as a new line character
318  for( ; (c != '\n') && (c != '\r') && ((!file.eof())); file.read(&c, 1))
319  nline.resize(nline.size() + 1, c);
320 
321  // Remove blank spaces & tabs from the beginning of the line
322  state = nline.find(" ", 0);
323  while(state != (int) string::npos && state == 0) {
324  nline.erase(state, 1);
325  state = nline.find(" ", state);
326  }
327 
328  state = nline.find("\t", 0);
329  while(state != (int) string::npos && state == 0) {
330  nline.erase(state, 1);
331  state = nline.find("\t", state);
332  }
333 
334  // If there is nothing to return, give back a NULL pointer ...
335  if(nline.size() == 0)
336  return NULL;
337 
338  // Otherwise, initialize the appropiate data structure,
339  // dump the data and return it
340  line = new char[nline.size() + 1];
341  strcpy(line, nline.c_str());
342  return line;
343 }

389 char* utils::trimLine(string nline) {
390  // This function is used to remove comments inbetween a biological sequence.
391  // Remove all content surrounded by ("") or ([]). It wans as well when a
392  // mismatch for these flags is found
393 
394  int pos, next;
395  static char *line;
396 
397  // Set-up lower and upper limit to look for comments inside of input string
398  pos = -1;
399 
400  // Identify comments inside of input sequence and remove it
401  while(true) {
402  pos = nline.find("\"", (pos + 1));
403 
404  // When there is not any more a comment inside of sequence,
405  // go out from this loop
406  if(pos == (int) string::npos)
407  break;
408 
409  // Look for closing flag
410  next = nline.rfind("\"", nline.size());
411 
412  // If a pair of comments flags '"' is found, remove everything inbetween
413  if((int) nline.find("\"", (pos + 1)) == next) {
414  nline.erase(pos, (next - pos + 1));
415  pos = -1;
416  }
417 
418  // If there is only one flag '"' for comments inside of sequence,
419  // user should be warned about that
420  if (pos == next) {
421  debug.report(ErrorCode::PossibleMissmatch);
422  return NULL;
423  }
424  }
425 
426  // Look for other kind of comments, in this case those with []
427  while(true) {
428  pos = -1;
429  next = -1;
430 
431  // Search for last opened bracket. It is supposed to be the first one for
432  // being close
433  while((pos = nline.find("[", (pos + 1))) != (int) string::npos)
434  next = pos;
435 
436  // If no opening bracket has been found.
437  // Check if there is any closing one
438  if (next == -1) {
439  // There are not any bracket in input string
440  if ((int) nline.find("]", 0) == (int) string::npos)
441  break;
442  // Otherwise, warn about the error
443  debug.report(ErrorCode::BracketsMissmatchFound);
444  return NULL;
445  }
446 
447  // Look for closest closing bracket to the opening one found
448  pos = nline.find("]", (next + 1));
449 
450  //If no closing bracket has been found. Warn about the mismatch
451  if (pos == (int) string::npos) {
452  debug.report(ErrorCode::BracketsMissmatchFound);
453  return NULL;
454  }
455 
456  // When both brackets have been found, remove comments inbetween them
457  nline.erase(next, (pos - next + 1));
458  }
459 
460  // Check if after removing all comments from input string there is still part
461  // of sequences or not
462  if(nline.size() == 0)
463  return NULL;
464 
465  // Initialize and store resulting sequence into an appropiate structure
466  line = new char[nline.size() + 1];
467  strcpy(line, &nline[0]);
468 
469  return line;
470 }

500 int utils::checkAlignmentType(int seqNumber, int residNumber, string *sequences) {
501 
502  int i, j, k, l, hitDNA, hitRNA, degenerate, gDNA, gRNA, extDNA, extRNA;
503  float ratioDNA, ratioRNA;
504  // Standard tables
505  char listRNA[11] = "AGCUNagcun";
506  char listDNA[11] = "AGCTNagctn";
507 
508  // Degenerate Nucleotides codes
509  char degeneratedCodes[21] = "MmRrWwSsYyKkVvHhDdBb";
510 
511  // For each sequences, this method locks at the 100 letters (excluding gaps).
512  // The method is able to distinguish between pure DNA/RNA nucleotides or those
513  // containing degenerate Nucleotide letters
514  for(i = 0, gDNA = 0, gRNA = 0, extDNA = 0, extRNA = 0; i < seqNumber; i++) {
515 
516  // Looks at the 100 letters (excluding gaps) while doesn's get the sequence's end
517  // When there are less than a 100 characters, break the loop before reaching that limit
518 
519  residNumber = (int) sequences[i].size();
520  for(j = 0, k = 0, hitDNA = 0, hitRNA = 0, degenerate = 0; j < residNumber; j++)
521  if(sequences[i][j] != '-' && sequences[i][j] != '.' && sequences[i][j] != '?') {
522  k++;
523 
524  /* Recognizes between DNA and RNA. */
525  for(l = 0; l < (int) strlen(listDNA); l++)
526  if(listDNA[l] == sequences[i][j])
527  hitDNA++;
528 
529  for(l = 0; l < (int) strlen(listRNA); l++)
530  if(listRNA[l] == sequences[i][j])
531  hitRNA++;
532 
533  for(l = 0; l < (int) strlen(degeneratedCodes); l++)
534  if(degeneratedCodes[l] == sequences[i][j])
535  degenerate++;
536  }
537 
538  // If input sequences have less than 95% of nucleotides, even when residues
539  // are treated with degenerated codes, consider the input file as containing
540  // amino-acidic sequences.
541  ratioDNA = float(degenerate + hitDNA)/k;
542  ratioRNA = float(degenerate + hitRNA)/k;
543 
544  if(ratioDNA < 0.95 && ratioRNA < 0.95)
545  return SequenceTypes::AA;
546 
547  // Identify precisely if nucleotides sequences are DNA/RNA strict or
548  // any degenerate code has been used in the sequence
549  else if(hitRNA > hitDNA && degenerate == 0)
550  gRNA++;
551  else if(hitRNA > hitDNA && degenerate != 0)
552  extRNA++;
553  else if(hitRNA < hitDNA && degenerate == 0)
554  gDNA++;
555  else if(hitRNA < hitDNA && degenerate != 0)
556  extDNA++;
557  }
558  // Return the datatype with greater values, considering always degenerate
559  // codes
560  if (extDNA != 0 && extDNA > extRNA)
561  return SequenceTypes::DNA | SequenceTypes::DEG;
562  else if (extRNA != 0 && extDNA < extRNA)
563  return SequenceTypes::RNA | SequenceTypes::DEG;
564  else if(gRNA > gDNA)
565  return SequenceTypes::RNA;
566  else
567  return SequenceTypes::DNA;
568 }
