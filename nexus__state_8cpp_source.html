183 bool NexusState::SaveAlignment(newAlignment* alignment, std::ostream* output, std::string* FileName)
184 {
185  /* Generate output alignment in NEXUS format setting only alignment block */
186 
187  int i, j, k, maxLongName = 0;
188  string *tmpMatrix;
189 
190  /* Check whether sequences in the alignment are aligned or not.
191  * Warn about it if there are not aligned. */
192  if (!alignment->isAligned) {
193  cerr << endl << "ERROR: Sequences are not aligned. Format (phylip) "
194  << "not compatible with unaligned sequences." << endl << endl;
195  return false;
196  }
197 
198  /* Allocate local memory for generating output alignment */
199  tmpMatrix = new string[alignment->sequenNumber];
200 
201  /* Depending on alignment orientation: forward or reverse. Copy directly
202  * sequence information or get firstly the reversed sequences and then
203  * copy it into local memory */
204  for(i = 0; i < alignment->sequenNumber; i++)
205  tmpMatrix[i] = (!Machine->reverse) ?
206  alignment->sequences[i] :
207  utils::getReverse(alignment->sequences[i]);
208 
209  /* Compute maximum sequences name length */
210  for(i = 0; (i < alignment->sequenNumber) && (!Machine->shortNames); i++)
211  maxLongName = utils::max(maxLongName, alignment->seqsName[i].size());
212 
213  /* Compute output file datatype */
214  alignment->getAlignmentType();
215 
216  /* Remove characters like ";" from input alignment information line */
217  while((int) alignment -> aligInfo.find(";") != (int) string::npos)
218  alignment ->aligInfo.erase(alignment -> aligInfo.find(";"), 1);
219 
220  /* Print Alignment header */
221  *output << "#NEXUS" << endl << "BEGIN DATA;" << endl << " DIMENSIONS NTAX="
222  << alignment->sequenNumber << " NCHAR=" << alignment->residNumber <<";" << endl;
223 
224  /* Print alignment datatype */
225  if (alignment->dataType == DNAType)
226  *output << "FORMAT DATATYPE=DNA INTERLEAVE=yes GAP=-";
227  else if (alignment->dataType == RNAType)
228  *output << "FORMAT DATATYPE=RNA INTERLEAVE=yes GAP=-";
229  else if (alignment->dataType == AAType)
230  *output << "FORMAT DATATYPE=PROTEIN INTERLEAVE=yes GAP=-";
231 
232  i = 0;
233  /* Using information from input alignment. Use only some tags. */
234  while((j = alignment ->aligInfo.find(" ", i)) != (int) string::npos) {
235 
236  if((alignment ->aligInfo.substr(i, j - i)).compare(0, 7, "MISSING") == 0 ||
237  (alignment ->aligInfo.substr(i, j)).compare(0, 7, "missing") == 0)
238  *output << " " << (alignment ->aligInfo.substr(i, j - i));
239 
240  else if((alignment ->aligInfo.substr(i, j)).compare(0, 9, "MATCHCHAR") == 0 ||
241  (alignment ->aligInfo.substr(i, j)).compare(0, 9, "matchchar") == 0)
242  *output << " " << (alignment ->aligInfo.substr(i, j - i));
243 
244  i = j + 1;
245  }
246  *output << ";" << endl;
247 
248  /* Print sequence name and sequence length */
249  for(i = 0; i < alignment->sequenNumber; i++)
250  *output << "[Name: " << setw(maxLongName + 4) << left << alignment->seqsName[i] << "Len: "
251  << alignment->residNumber << "]" << endl;
252  *output << endl << "MATRIX" << endl;
253 
254  /* Print alignment itself. Sequence name and 50 residues blocks */
255  for(j = 0; j < alignment->residNumber; j += 50) {
256  for(i = 0; i < alignment->sequenNumber; i++) {
257  *output << setw(maxLongName + 4) << left << alignment->seqsName[i];
258  for(k = j; k < (j + 50) && k < alignment->residNumber; k += 10)
259  *output << " " << alignment->sequences[i].substr(k, 10);
260  *output << endl;
261  }
262  *output << endl;
263  }
264  *output << ";" << endl << "END;" << endl;
265 
266  /* Deallocate local memory */
267  delete [] tmpMatrix;
268 
269  return true;
