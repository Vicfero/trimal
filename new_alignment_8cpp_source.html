292 newAlignment *newAlignment::getTranslationCDS(int newResidues, int newSequences, int *ColumnsToKeep, string *oldSeqsName, sequencesMatrix *seqMatrix,
293  newAlignment *ProtAlig) {
294 
295  string *matrixAux;
296  newAlignment *newAlig;
297  int i, j, k, l, oldResidues;
298  int *mappedSeqs, *tmpSequence, *selectedRes;
299 
300  /* ***** ***** ***** ***** ***** ***** ***** ***** */
301  /* Map the selected protein sequences to the input
302  * coding sequences */
303  mappedSeqs = new int[newSequences];
304  for(i = 0; i < sequenNumber; i++)
305  for(j = 0; j < newSequences; j++)
306  if(!seqsName[i].compare(oldSeqsName[j])) {
307  mappedSeqs[j] = i;
308  break;
309  }
310  /* ***** ***** ***** ***** ***** ***** ***** ***** */
311 
312  /* ***** ***** ***** ***** ***** ***** ***** ***** */
313  /* Get the original newAlignment size as well the original
314  * selected/non-selected columns */
315  oldResidues = seqMatrix -> getResidNumber();
316  selectedRes = new int[oldResidues];
317 
318  for(i = 0; i < oldResidues; i++)
319  selectedRes[i] = i;
320 
321  for(j = 0; j < ColumnsToKeep[0]; j++)
322  selectedRes[j] = -1;
323 
324  for(i = 0; i < newResidues - 1; i++)
325  for(j = ColumnsToKeep[i] + 1; j < ColumnsToKeep[i+1]; j++)
326  selectedRes[j] = -1;
327 
328  for(j = ColumnsToKeep[newResidues - 1] + 1; j < oldResidues; j++)
329  selectedRes[j] = -1;
330  /* ***** ***** ***** ***** ***** ***** ***** ***** */
331 
332  /* ***** ***** ***** ***** ***** ***** ***** ***** */
333  /* We allocate memory to save the columns selected */
334  matrixAux = new string[newSequences];
335  tmpSequence = new int[oldResidues];
336 
337  /* Using the information about which residues for each
338  * sequence was selected by others function, we process
339  * these residues to recover the corresponding codons */
340  for(i = 0; i < newSequences; i++)
341  if(seqMatrix -> getSequence(oldSeqsName[i], tmpSequence)) {
342  for(j = 0; j < oldResidues; j++) {
343  if((selectedRes[j] != -1) && (tmpSequence[j] != 0)) {
344 
345  for(k = 3 * (tmpSequence[j] - 1), l = 0; l < 3; k++, l++) {
346  /* Check whether the nucleotide sequences end has been reached or not.
347  * If it has been reached, complete backtranslation using indetermination
348  * symbols 'N' */
349  if((int) sequences[mappedSeqs[i]].length() > k)
350  matrixAux[i].resize(matrixAux[i].size() + 1, sequences[mappedSeqs[i]][k]);
351  else
352  matrixAux[i].resize(matrixAux[i].size() + 1, 'N');
353  }
354  } else if(selectedRes[j] != -1) {
355  matrixAux[i].resize(matrixAux[i].size() + 3, '-');
356  }
357  }
358  /* If there is any problems with a sequence then
359  * the function returns an error */
360  } else {
361  return NULL;
362  }
363 
364  newAlig = new newAlignment(*this);
365  newAlig -> aligInfo = "";
366  newAlig -> seqsInfo = NULL;
367 
368  newAlig -> sequenNumber = newSequences;
369  newAlig -> residNumber = newResidues * 3;
370 
371  newAlig -> sequences = new string[newSequences];
372  newAlig -> seqsName = new string[newSequences];
373 
374  newAlig -> dataType = DNAType;
375  newAlig -> isAligned = true;
376 
377 // newAlig -> reverse = ProtAlig -> getReverseFlag();
378 // newAlig -> OldResidues = oldResidues * 3; TODO?
379  newAlig -> residuesNumber = NULL;
380  newAlig -> saveSequences = NULL;
381  newAlig -> saveResidues = NULL;
382 
383 // newAlig -> ghWindow = 0;
384 // newAlig -> shWindow = 0;
385 
386  newAlig -> Cleaning -> blockSize = ProtAlig -> getBlockSize();
387 
388  newAlig -> identities = NULL;
389 
390  newAlig -> saveResidues = new int[residNumber];
391 
392 
393 
394  /* ***** ***** ***** ***** ***** ***** ***** ***** */
395 
396  /* ***** ***** ***** ***** ***** ***** ***** ***** */
397  /* Deallocated auxiliar memory */
398  delete [] matrixAux;
399  delete mappedSeqs;
400  delete tmpSequence;
401  delete selectedRes;
402  /* ***** ***** ***** ***** ***** ***** ***** ***** */
403 
404  /* ***** ***** ***** ***** ***** ***** ***** ***** */
405  /* Return the new newAlignment reference */
406  return newAlig;
407  /* ***** ***** ***** ***** ***** ***** ***** ***** */
408 }

610 bool newAlignment::prepareCodingSequence(bool splitByStopCodon, bool ignStopCodon,\
611  newAlignment *proteinAlig) {
612 
613  if (getAlignmentType() == AAType) {
614  cerr << endl << "ERROR: Check input CDS file. It seems to content protein "

619  bool warning = false;
620  size_t found;
621  int i;
622 
623  /* New code: We now care about the presence of wildcards/indeterminations
624  * characters such as 'X' or 'B' into protein sequences as well as about the
625  * presence of Selenocysteines ('U') or Pyrrolysines ('O'). It only works, by
626  * now, with the universal genetic code */
627  int *protSeqsLengths, numbProtSeqs, current_prot;
628  string *protSeqsNames, *protSequences;
629  char aminoAcid;

637 
638  /* Check read sequences are real DNA/RNA */
639 
640  for(i = 0; i < sequenNumber; i++) {
641 
642  /* Get protein sequence to compare against any potential stop codon in the
643  * coding sequence. If there is not protein sequence for current coding
644  * sequence, skip its analysis */
645  for(current_prot = 0; current_prot < numbProtSeqs; current_prot++)
646  if(protSeqsNames[current_prot] == seqsName[i])
647  break;
648  if(current_prot == numbProtSeqs)
649  continue;
650 
651  if(sequences[i].find("-") != string::npos) {
