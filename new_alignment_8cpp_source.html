406 void newAlignment::calculateSeqIdentity(void) {
407 
408  int i, j, k, hit, dst;
409  char indet;
410 
411  /* Depending on alignment type, indetermination symbol will be one or other */
412  indet = getAlignmentType() & SequenceTypes::AA ? 'X' : 'N';
413 
414  /* Create identities matrix to store identities scores */
415  identities = new float*[sequenNumber];
416 
417  /* For each seq, compute its identity score against the others in the MSA */
418  for(i = 0; i < sequenNumber; i++) {
419  identities[i] = new float[sequenNumber];
420 
421  /* It's a symmetric matrix, copy values that have been already computed */
422  for(j = 0; j < i; j++)
423  identities[i][j] = identities[j][i];
424  identities[i][i] = 0;
425 
426  /* Compute identity scores for the current sequence against the rest */
427  for(j = i + 1; j < sequenNumber; j++) {
428  for(k = 0, hit = 0, dst = 0; k < residNumber; k++) {
429  /* If one of the two positions is a valid residue,
430  * count it for the common length */
431  if(((sequences[i][k] != indet) && (sequences[i][k] != '-')) ||
432  ((sequences[j][k] != indet) && (sequences[j][k] != '-'))) {
433  dst++;
434  /* If both positions are the same, count a hit */
435  if(sequences[i][k] == sequences[j][k])
436  hit++;
437  }
438  }
439 
440  /* Identity score between two sequences is the ratio of identical residues
441  * by the total length (common and no-common residues) among them */
442  identities[i][j] = (float) hit/dst;
443  }
444  }

448  /* Raw approximation of sequence identity computation designed for reducing
449  * comparisons for huge alignemnts */
450 
451  int i, j, k, hit;
452 
453  /* Create identities matrix to store identities scores */
454  identities = new float*[sequenNumber];
455

457  for(i = 0; i < sequenNumber; i++) {

459 
460  /* It's a symmetric matrix, copy values that have been already computed */
461  for(j = 0; j < i; j++)
462  identities[i][j] = identities[j][i];
463  identities[i][i] = 0;
464 
465  /* Compute identity score between the selected sequence and the others */
466  for(j = i + 1; j < sequenNumber; j++) {

468  /* If both positions are the same, count a hit */

471  }

473  * alignment length */
474  identities[i][j] = (float) hit/residNumber;
475  }

480 void newAlignment::calculateSeqOverlap(void) {
481  /* Compute the overlap between sequences taken each of them as the reference
482  * to compute such scores. It will lead to a non-symmetric matrix. */
483 
484  int i, j, k, shared, referenceLength;
485  char indet;
486 
487  /* Depending on alignment type, indetermination symbol will be one or other */
488  indet = getAlignmentType() & SequenceTypes::AA ? 'X' : 'N';
489 
490  /* Create overlap matrix to store overlap scores */
491  overlaps = new float*[sequenNumber];
492 
493  /* For each seq, compute its overlap score against the others in the MSA */
494  for(i = 0; i < sequenNumber; i++) {
495  overlaps[i] = new float[sequenNumber];
496 
497  for(j = 0; j < sequenNumber; j++) {
498  for(k = 0, shared = 0, referenceLength = 0; k < residNumber; k++) {
499  /* If there a valid residue for the reference sequence, then see if
500  * there is a valid residue for the other sequence. */
501  if((sequences[i][k] != indet) && (sequences[i][k] != '-')) {
502  referenceLength++;
503  if ((sequences[j][k] != indet) && (sequences[j][k] != '-'))
504  shared++;
505  }
506  }
507  /* Overlap score between two sequences is the ratio of shared valid
508  * residues divided by the sequence length taken as reference. The
509  * overlaps matrix, therefore, will be not symmetric. */
510  overlaps[i][j] = (float) shared/referenceLength;
511  }
512  }
