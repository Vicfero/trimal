22 bool PhylipPamlState::SaveAlignment(newAlignment* alignment, std::ostream* output, std::string* FileName)
23 {
24  /* Generate output alignment in PHYLIP format compatible with PAML program */
25 
26  int i, maxLongName;
27  string *tmpMatrix;
28 
29  /* Check whether sequences in the alignment are aligned or not.
30  * Warn about it if there are not aligned. */
31  if (!alignment->isAligned) {
32  cerr << endl << "ERROR: Sequences are not aligned. Format (phylip) "
33  << "not compatible with unaligned sequences." << endl << endl;
34  return false;
35  }
36 
37  /* Allocate local memory for generating output alignment */
38  tmpMatrix = new string[alignment->sequenNumber];
39 
40  /* Depending on alignment orientation: forward or reverse. Copy directly
41  * sequence information or get firstly the reversed sequences and then
42  * copy it into local memory */
43  for(i = 0; i < alignment->sequenNumber; i++)
44  tmpMatrix[i] = (!Machine->reverse) ?
45  alignment->sequences[i] :
46  utils::getReverse(alignment->sequences[i]);
47 
48  /* Depending on if short name flag is activated (limits sequence name up to
49  * 10 characters) or not, get maximum sequence name length */
50  maxLongName = PHYLIPDISTANCE;
51  for(i = 0; (i < alignment->sequenNumber) && (!Machine->shortNames); i++)
52  maxLongName = utils::max(maxLongName, alignment->seqsName[i].size());
53 
54  /* Generating output alignment */
55  /* First Line: Sequences Number & Residued Number */
56  *output << " " << alignment->sequenNumber << " " << alignment->residNumber << endl;
57 
58  /* Print alignment */
59  /* Print sequences name follow by the sequence itself in the same line */
60  for(i = 0; i < alignment->sequenNumber; i++)
61  *output << setw(maxLongName + 3) << left << alignment->seqsName[i].substr(0, maxLongName)
62  << alignment->sequences[i] << endl;
63  *output << endl;
64 
65  /* Deallocate local memory */
66  delete [] tmpMatrix;
67 
68  return true;
69 }
