195 int checkArguments(ReadWriteMS* machine, std::vector<std::string>* inFiles, std::vector<std::string>* outFormats, std::string* outPattern)
196 {
197  int returnValue = 0;
198  if (inFiles->size() == 0)
199  {
200  cerr << "ERROR: At least one input file must be provided" << endl;
201  returnValue = 1;
202  }
203  if (*outPattern == "")
204  {
205  if (inFiles->size() == 1 && outFormats->size() == 1 && !(machine->format || machine->info || machine->type))
206  machine->hasOutputFile = false;
207  else if (outFormats->size() != 0)
208  {
209  cerr << "ERROR: Terminal output option not compatible with information printing (-info | -format | -type)" << endl
210  << "Provide an output format or disable information printing." << endl;
211  returnValue = 1;
212  }
213  }
214  else if (outFormats->size() == 0)
215  {
216  cerr << "ERROR: At least one output format must be provided" << endl;
217  returnValue = 1;
218  }
219  return returnValue;
