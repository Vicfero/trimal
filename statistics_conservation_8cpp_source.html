115 | void statisticsConservation::calculateMatrixIdentity(char **, int, int) |
116 | |
117 | This method computes the matrix identity between all the sequences in the alignment. |
118 | |
119 ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++*/
120 
121 void statisticsConservation::calculateMatrixIdentity(string *alignmentMatrix) {
122 
123  char indet;
124  int i, j, k, sum, length;
125 
126  if(dataType == AAType)
127  indet = 'X';
128  else
129  indet = 'N';
130 
131  /* For each sequences' pair */
132  for(i = 0; i < sequences; i++) {
133  for(j = i + 1; j < sequences; j++) {
134 
135  /* For each position in the alignment of that pair than we are processing */
136  for(k = 0, sum = 0, length = 0; k < columns; k++) {
137 
138  /* If we find a element that is not a gap or an X aminoacid in the first sequence of the pair */
139  if((alignmentMatrix[i][k] != '-') && (alignmentMatrix[i][k] != indet)) {
140 
141  /* If we also find a valid element in the second sequence */
142  if((alignmentMatrix[j][k] != '-') && (alignmentMatrix[j][k] != indet))
143 
144  /* If the two valid elements are the same increase the sum */
145  if(alignmentMatrix[j][k] == alignmentMatrix[i][k])
146  sum++;
147 
148  /* Increase the length of the sequence free of gaps and X elements */
149  length++;
150  }
151 
152  /* If the first processed element is invalid and in the second we find a valid element increase the length of
153  the sequence free of gaps and X elements */
154  else if((alignmentMatrix[j][k] != '-') && (alignmentMatrix[j][k] != indet))
155  length++;
156  }
157 
158  /* Calculate the value of matrixidn for columns j and i */
159  matrixIdentity[j][i] = (100.0 - ((float) sum/ length) * 100.0);
160  matrixIdentity[i][j] = matrixIdentity[j][i];
161  }
162  }
163 }

166 | bool statisticsConservation::calculateVectors(char **, int *) |
167 | |
168 | This method computes the distance between pairs for each column in the alignment. |
169 | |
170 ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++*/
171 
172 bool statisticsConservation::calculateVectors(string *alignmentMatrix, int *gaps) {
173 
174  char indet;
175  int i, j, k;
176  float num, den;
177 
178  if(dataType == AAType)
179  indet = 'X';
180  else
181  indet = 'N';
182 
183  /* A conservation matrix must be defined. If not, return false */
184  if(simMatrix == NULL)
185  return false;
186 
187  /* For each column calculate the Q value and the MD value using an equation */
188  for(i = 0; i < columns; i++) {
189  /* For each AAs/Nucleotides' pair in the column we compute its distance */
190  for(j = 0, num = 0, den = 0; j < sequences; j++) {
191  /* We don't compute the distant if the first element is a indeterminate (X) or a gap (-) element. */
192  if((alignmentMatrix[j][i] != '-') && (alignmentMatrix[j][i] != indet))
193  for(k = j + 1; k < sequences; k++)
194  /* We don't compute the distant between the pair if the second element is a indeterminate or a gap element */
195  if((alignmentMatrix[k][i] != '-') && (alignmentMatrix[k][i] != indet)) {
196  /* We use the identity value for the two pairs and its distance based on similarity matrix's value. */
197  num += matrixIdentity[j][k] * simMatrix -> getDistance(alignmentMatrix[j][i], alignmentMatrix[k][i]);;
198  den += matrixIdentity[j][k];
199  }
200  }
201  /* If we are procesing a column with only one AA/nucleotide, the denominator is 0 and we don't execute the division
202  and we set the Q[i] value to 0. */
203  Q[i] = (den == 0) ? 0 : num / den;
204  MDK[i] = (float) exp(-Q[i]);
205 
206  /* If the column has 80% or more gaps then we set its conservation value to 0 */
207  if(gaps != NULL)
208  if(((float) gaps[i] / sequences) >= 0.8) MDK[i] = 0;
209 
210  /* If the MDK value is more than 1, we normalized this value to 1. */
211  if(MDK[i] > 1) MDK[i] = 1;
212  }
213 
214  return true;
215 }

217 /*++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
218 | bool statisticsConservation::applyWindow(int) |
219 | |
220 | This method computes for each column's alignment its conservationwindows' value. For this purpose, the method |
221 | uses the values that previously has been calculated and the window's size value. |
222 | |
223 ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++*/
224 
225 bool statisticsConservation::applyWindow(int _halfWindow) {
226 
227  int i, j, window;
228 
229  /* If one of this conditions is true, we return FALSE: */
230  /* .- If already exists a previously calculated vector for this window size */
231  /* .- If mediumWinSize value is greater than 1/4 of alignment length */
232  if((halfWindow == _halfWindow) || (_halfWindow > columns/4))
233  return false;
234 
235  halfWindow = _halfWindow;
236  window = 2 * halfWindow + 1;
237 
238  /* Do the average window calculations */
239  for(i = 0; i < columns; i++) {
240  for(j = i - halfWindow; j <= i + halfWindow; j++) {
241  if(j < 0) MDK_Window[i] += MDK[-j];
242  else if(j >= columns) MDK_Window[i] += MDK[((2 * columns - j) - 2)];
243  else MDK_Window[i] += MDK[j];
244  }
245 
246  /* Calculate the similiraty value for the i column */
247  MDK_Window[i] = MDK_Window[i] / (float) window;
248  }
249  return true;
250 }

308 /*++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
309 | float statisticsConservation::calcCutPoint(float baseLine, float conservationPct) |
310 | |
311 | This method computes and selects the cut point value based on alignment's conservation. For this purpose, this |
312 | select the minimum conservation value bewteen the conservationPct -a conservation value- and the conservation |
313 | value allows us conserve the columns' number associated to baseline. |
314 | |
315 ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++*/
316 
317 float statisticsConservation::calcCutPoint(float baseLine, float conservationPct) {
318 
319  float cutBL, cutCons, *vectAux;
320  int i;
321 
322  /* Allocate memory */
323  vectAux = new float[columns];
324 
325  /* Sort a copy of the MDK_Window vector, and take the value of the column that marks the % baseline */
326  utils::copyVect(MDK_Window, vectAux, columns);
327  utils::quicksort(vectAux, 0, columns-1);
328 
329  for(i = columns - 1; i >= 0; i--)
330  if(vectAux[i] < conservationPct) break;
331  cutCons = vectAux[i];
332 
333  cutBL = vectAux[(int) ((float)(columns - 1) * (100.0 - baseLine)/100.0)];
334 
335  /* Deallocate memory */
336  delete[] vectAux;
337 
338  /* Return the minimum of the baseLine cut and conservationPct value */
339  return (cutBL < cutCons ? cutBL : cutCons);
340 }
