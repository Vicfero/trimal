148

150  maxLongName = PHYLIPDISTANCE;

152  cerr << endl << "WARNING: Original sequence header will be cutted by "

156  for(i = 0; i < alignment->sequenNumber; i++) {

159  else if (alignment->seqsInfo != NULL)
